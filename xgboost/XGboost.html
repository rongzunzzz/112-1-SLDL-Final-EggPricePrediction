import math
import torch
def MSPE(pred, labels):

    # l1_or = torch.nn.L1Loss(reduction="none")

    # pred = pred.to(device)
    # labels = labels.to(device)
    diff = pred - labels
    x = diff/labels
    result = np.mean(x.pow(2))

    return result
